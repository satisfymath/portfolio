------------------------------------------------------------

import Lenis from "@studio-freight/lenis";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

export function initLenis() {
  const lenis = new Lenis({
    lerp: 0.10,
    smoothWheel: true,
    smoothTouch: false, // móvil: evita sensación rara; puedes cambiarlo luego
  });

  lenis.on("scroll", () => ScrollTrigger.update());

  gsap.ticker.add((time) => {
    lenis.raf(time * 1000);
  });

  gsap.ticker.lagSmoothing(0);

  return lenis;
}

7) Componente clave: VideoScrub (controla mp4 con scroll)
------------------------------------------------------------

Archivo: /src/app/components/VideoScrub.tsx

import { useEffect, useRef } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import { prefersReducedMotion } from "../lib/reducedMotion";

gsap.registerPlugin(ScrollTrigger);

type Props = {
  src: string;               // "/media/video/hero_a.mp4"
  trigger: string;           // selector del section (ej: "#hero")
  start?: string;            // "top top"
  end?: string;              // "+=1200"
  className?: string;
  poster?: string;
};

export function VideoScrub({
  src,
  trigger,
  start = "top top",
  end = "+=1200",
  className,
  poster,
}: Props) {
  const ref = useRef<HTMLVideoElement | null>(null);

  useEffect(() => {
    const video = ref.current;
    if (!video) return;

    // iOS/autoplay safe
    video.muted = true;
    video.playsInline = true;
    video.preload = "auto";

    // reduced motion: no scrub, solo loop
    if (prefersReducedMotion()) {
      video.loop = true;
      video.autoplay = true;
      video.play().catch(() => {});
      return;
    }

    let st: ScrollTrigger | null = null;

    const onLoaded = () => {
      // importante: pausar para que scroll controle currentTime
      video.pause();
      video.currentTime = 0;

      // GSAP animando currentTime según el progreso del scroll
      const tl = gsap.timeline({ defaults: { ease: "none" } });
      tl.to(video, { currentTime: video.duration }, 0);

      st = ScrollTrigger.create({
        trigger,
        start,
        end,
        scrub: true,
        animation: tl,
        invalidateOnRefresh: true,
      });

      ScrollTrigger.refresh();
    };

    video.addEventListener("loadedmetadata", onLoaded);

    return () => {
      video.removeEventListener("loadedmetadata", onLoaded);
      st?.kill();
    };
  }, [src, trigger, start, end]);

  return (
    <video
      ref={ref}
      className={className}
      src={src}
      poster={poster}
      muted
      playsInline
      preload="auto"
    />
  );
}

import { useEffect, useState } from "react";

export function HUD() {
  const [p, setP] = useState(0);

  useEffect(() => {
    const onScroll = () => {
      const h = document.documentElement;
      const max = h.scrollHeight - h.clientHeight;
      setP(max > 0 ? (h.scrollTop / max) : 0);
    };
    onScroll();
    window.addEventListener("scroll", onScroll, { passive: true });
    return () => window.removeEventListener("scroll", onScroll);
  }, []);

  return (
    <div style={{ position: "fixed", top: 0, left: 0, right: 0, height: 2, zIndex: 60 }}>
      <div style={{
        height: "100%",
        width: `${p * 100}%`,
        background: "var(--accent)",
        opacity: 0.65
      }} />
    </div>
  );
}

16) ⚠️ COMPATIBILIDAD iOS SAFARI (CRÍTICO)
------------------------------------------------------------

iOS Safari tiene restricciones estrictas para videos. Sin estos ajustes,
los videos aparecerán en negro o no responderán al scroll.

16.1 Atributos HTML obligatorios para <video>:

<video
  ref={videoRef}
  src={videoSrc}
  muted                              // OBLIGATORIO para autoplay/scrub
  playsInline                        // OBLIGATORIO: evita fullscreen automático
  autoPlay={false}                   // Evita conflictos con scrub
  preload="auto"                     // Carga anticipada de frames
  {...{ 'webkit-playsinline': 'true' }}  // Legacy iOS Safari
/>

16.2 Forzar carga en iOS:

iOS a veces no carga metadatos hasta que se interactúa.
Forzar con video.load() si readyState < 1:

useEffect(() => {
  const video = videoRef.current;
  if (!video) return;
  
  video.muted = true;
  video.playsInline = true;
  video.preload = 'auto';
  
  // CRÍTICO: Forzar carga en iOS
  if (video.readyState < 1) {
    video.load();
  }
  // ...
}, []);

16.3 "Kickstart" del video (anti pantalla negra):

iOS no renderiza frames hasta que el video se haya "reproducido" al menos una vez.
Hack: llamar play() y luego pause() inmediatamente para activar el decodificador:

const onLoadedMetadata = () => {
  // iOS HACK: Kickstart - activar decodificador
  const playPromise = video.play();
  if (playPromise !== undefined) {
    playPromise.then(() => {
      console.log('Kickstart exitoso');
      video.pause();
      video.currentTime = 0;
      initializeScrollTrigger();
    }).catch((error) => {
      console.warn('Kickstart falló:', error);
      // Continuar de todas formas
      video.pause();
      initializeScrollTrigger();
    });
  } else {
    video.pause();
    initializeScrollTrigger();
  }
};

17) CODIFICACIÓN DE VIDEO PARA SCRUBBING (GOP bajo)
------------------------------------------------------------

El "scrubbing" (controlar video con scroll) requiere saltar a cualquier
frame instantáneamente. Los videos normales tienen Keyframes cada ~250 frames
lo que causa "saltos" o "stuttering" al hacer scrub.

17.1 Solución: GOP (Group of Pictures) bajo

Recodifica con Keyframe cada 4 frames (-g 4):

ffmpeg -y -i input.mp4 \
  -c:v libopenh264 \       # O libx264 si está disponible
  -g 4 \                   # Keyframe cada 4 frames (CRUCIAL)
  -keyint_min 4 \          # Mínimo entre keyframes
  -pix_fmt yuv420p \       # Compatible con todos los navegadores
  -movflags +faststart \   # Metadatos al inicio (streaming)
  -an \                    # Sin audio (innecesario para scrub)
  output.mp4

17.2 Verificar codec del video:

ffprobe -v quiet -show_streams -select_streams v:0 video.mp4 | grep -E "codec_name|profile|level|pix_fmt"

Resultado esperado:
- codec_name=h264
- profile=Baseline (o Main)
- pix_fmt=yuv420p

17.3 Verificar posición del moov atom (faststart):

ffprobe -v trace video.mp4 2>&1 | grep -i "moov\|ftyp" | head -5

El "moov" debe aparecer ANTES que "mdat" para streaming rápido.

19) COMPONENTE VideoBackgroundSection (patrón sticky)
------------------------------------------------------------

Patrón para video de fondo que se queda fijo mientras el contenido
pasa encima. Usa position: sticky en lugar de fixed.

import { useEffect, useRef, type ReactNode } from 'react';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { prefersReducedMotion } from '../lib/reducedMotion';

gsap.registerPlugin(ScrollTrigger);

interface Props {
  id: string;
  videoSrc: string;
  children: ReactNode;
}

export function VideoBackgroundSection({ id, videoSrc, children }: Props) {
  const sectionRef = useRef<HTMLElement | null>(null);
  const videoRef = useRef<HTMLVideoElement | null>(null);
  const scrollTriggerRef = useRef<ScrollTrigger | null>(null);
  const timelineRef = useRef<gsap.core.Timeline | null>(null);
  const videoId = videoSrc.split('/').pop() || 'unknown';

  useEffect(() => {
    const section = sectionRef.current;
    const video = videoRef.current;
    if (!section || !video) return;

    console.log(`[VideoBg:${videoId}] Init`);

    // Setup iOS-compatible
    video.muted = true;
    video.playsInline = true;
    video.preload = 'auto';
    
    if (video.readyState < 1) {
      video.load();
    }

    if (prefersReducedMotion()) {
      video.loop = true;
      video.autoplay = true;
      video.play().catch(() => {});
      return;
    }

    const onLoadedMetadata = () => {
      console.log(`[VideoBg:${videoId}] Loaded: ${video.duration.toFixed(2)}s, ${video.videoWidth}x${video.videoHeight}`);
      
      // iOS Kickstart
      const playPromise = video.play();
      if (playPromise !== undefined) {
        playPromise.then(() => {
          console.log(`[VideoBg:${videoId}] Kickstart OK`);
          video.pause();
          video.currentTime = 0;
          initScrollTrigger();
        }).catch((err) => {
          console.warn(`[VideoBg:${videoId}] Kickstart failed:`, err);
          video.pause();
          initScrollTrigger();
        });
      } else {
        video.pause();
        initScrollTrigger();
      }
    };

    const initScrollTrigger = () => {
      if (timelineRef.current) timelineRef.current.kill();
      if (scrollTriggerRef.current) scrollTriggerRef.current.kill();

      timelineRef.current = gsap.timeline({ defaults: { ease: 'none' } });
      timelineRef.current.to(video, { currentTime: video.duration }, 0);

      scrollTriggerRef.current = ScrollTrigger.create({
        trigger: section,
        start: 'top top',
        end: 'bottom bottom',
        scrub: 1,  // Suavizado de 1 segundo
        animation: timelineRef.current,
        invalidateOnRefresh: true,
        onUpdate: (self) => {
          const pct = Math.round(self.progress * 100);
          if (pct % 25 === 0) {
            console.log(`[${videoId}] ${pct}%`);
          }
        },
      });
      ScrollTrigger.refresh();
    };

    const onError = (e: Event) => {
      const target = e.target as HTMLVideoElement;
      console.error(`[VideoBg:${videoId}] ERROR:`, target.error);
    };

    video.addEventListener('loadedmetadata', onLoadedMetadata);
    video.addEventListener('error', onError);
    
    if (video.readyState >= 1) {
      onLoadedMetadata();
    }

    return () => {
      video.removeEventListener('loadedmetadata', onLoadedMetadata);
      video.removeEventListener('error', onError);
      if (scrollTriggerRef.current) scrollTriggerRef.current.kill();
      if (timelineRef.current) timelineRef.current.kill();
    };
  }, [videoSrc, videoId]);

  return (
    <section id={id} ref={sectionRef} style={{ position: 'relative' }}>
      {/* Video sticky */}
      <div style={{
        position: 'sticky',
        top: 0,
        left: 0,
        width: '100%',
        height: '100vh',
        overflow: 'hidden',
        zIndex: 0,
      }}>
        <video
          ref={videoRef}
          src={videoSrc}
          muted
          playsInline
          autoPlay={false}
          preload="auto"
          {...{ 'webkit-playsinline': 'true' }}
          style={{
            position: 'absolute',
            top: 0,
            left: 0,
            width: '100%',
            height: '100%',
            objectFit: 'cover',
          }}
        />
      </div>

      {/* Contenido sobre el video */}
      <div style={{
        position: 'relative',
        marginTop: '-100vh',
        zIndex: 1,
      }}>
        {children}
      </div>
    </section>
  );
}

20) LENIS EN MÓVIL (smoothTouch)
------------------------------------------------------------

Lenis puede causar problemas en móvil. Configuración recomendada:

20.1 Deshabilitar en móvil (recomendado):

export function initLenis() {
  // Detectar móvil
  const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);
  
  if (isMobile) {
    console.log('[Lenis] Disabled - using native scroll');
    return null;  // Usar scroll nativo
  }
  
  const lenis = new Lenis({
    lerp: 0.10,
    smoothWheel: true,
    smoothTouch: false,
  });
  
  lenis.on('scroll', () => ScrollTrigger.update());
  gsap.ticker.add((time) => lenis.raf(time * 1000));
  gsap.ticker.lagSmoothing(0);
  
  return lenis;
}

20.2 Cleanup en App.tsx:

export function destroyLenis(lenis: Lenis | null) {
  if (lenis?.destroy) {
    lenis.destroy();
  }
}

// En useEffect:
useEffect(() => {
  const lenis = initLenis();
  return () => destroyLenis(lenis);
}, []);

21) ERRORES COMUNES Y SOLUCIONES
------------------------------------------------------------

21.1 Video negro en iOS:
- Agregar webkit-playsinline="true"
- Implementar kickstart play/pause
- Verificar que video tiene muted y playsInline

21.2 Video "salta" o tiene stuttering:
- Recodificar con GOP bajo (-g 4)
- Aumentar scrub value (scrub: 1 en vez de scrub: 0.5)
- Verificar que video tiene faststart (movflags)

21.3 Video no carga en móvil:
- Llamar video.load() explícitamente
- Verificar preload="auto"
- Revisar tamaño del archivo (< 5MB recomendado)

21.4 ScrollTrigger no funciona:
- Verificar que trigger element existe en DOM
- Llamar ScrollTrigger.refresh() después de setup
- Agregar invalidateOnRefresh: true

21.5 Lenis desincronizado con ScrollTrigger:
- Verificar lenis.on('scroll', () => ScrollTrigger.update())
- Asegurar gsap.ticker.lagSmoothing(0)

21.6 Video duration es NaN o 0:
- Esperar evento 'loadedmetadata' antes de usar duration
- Verificar que video existe en la ruta correcta

24) TAMAÑOS RECOMENDADOS DE VIDEO
------------------------------------------------------------

Para web responsive con scrubbing:

- Resolución: 1024x1024 o 1280x720 (máximo)
- Duración: 4-8 segundos (ideal para scrub)
- Framerate: 15-24 fps (suficiente para scrub)
- Bitrate: 1.5-2.5 Mbps
- Tamaño final: < 5MB por video
- Codec: H.264 Baseline o Main
- Pixel format: yuv420p

Si el video es muy pesado:
- Reducir resolución
- Reducir framerate
- Reducir duración
- Aumentar compresión (menor calidad visual)

25) SCRIPT DE RECODIFICACIÓN BATCH
------------------------------------------------------------

#!/bin/bash
# recodificar_para_scrub.sh

INPUT_DIR="./videos_originales"
OUTPUT_DIR="./public/media/video"

mkdir -p "$OUTPUT_DIR"

for file in "$INPUT_DIR"/*.mp4; do
  filename=$(basename "$file")
  echo "Procesando: $filename"
  
  ffmpeg -y -i "$file" \
    -c:v libopenh264 \
    -g 4 \
    -keyint_min 4 \
    -pix_fmt yuv420p \
    -movflags +faststart \
    -an \
    "$OUTPUT_DIR/$filename"
done

echo "✓ Todos los videos procesados"
