# llm_branding_lenis_GSAP_guide.txt
# Objetivo
Montar una landing Brutalist Precision Noir (Vite + React + TS) con:
- Smooth scroll (Lenis)
- Animaciones (GSAP + ScrollTrigger)
- 3 videos mp4 (97 frames) que se “scrubbean” con el scroll (control de currentTime)
- Tipografías:
  - Orbitron (títulos / marca)
  - Raleway Dots (números dinámicos / índices)
  - Google Sans Code (bloques de “código” / microtexto técnico)

5) Lenis (smooth scroll) + sincronización con GSAP
------------------------------------------------------------

Archivo: /src/app/lib/lenis.ts

import Lenis from "@studio-freight/lenis";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

export function initLenis() {
  const lenis = new Lenis({
    lerp: 0.10,
    smoothWheel: true,
    smoothTouch: false, // móvil: evita sensación rara; puedes cambiarlo luego
  });

  lenis.on("scroll", () => ScrollTrigger.update());

  gsap.ticker.add((time) => {
    lenis.raf(time * 1000);
  });

  gsap.ticker.lagSmoothing(0);

  return lenis;
}

export function prefersReducedMotion(): boolean {
  if (typeof window === "undefined") return false;
  return window.matchMedia("(prefers-reduced-motion: reduce)").matches;
}

7) Componente clave: VideoScrub (controla mp4 con scroll)
------------------------------------------------------------

Archivo: /src/app/components/VideoScrub.tsx

import { useEffect, useRef } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import { prefersReducedMotion } from "../lib/reducedMotion";

gsap.registerPlugin(ScrollTrigger);

type Props = {
  src: string;               // "/media/video/hero_a.mp4"
  trigger: string;           // selector del section (ej: "#hero")
  start?: string;            // "top top"
  end?: string;              // "+=1200"
  className?: string;
  poster?: string;
};

export function VideoScrub({
  src,
  trigger,
  start = "top top",
  end = "+=1200",
  className,
  poster,
}: Props) {
  const ref = useRef<HTMLVideoElement | null>(null);

  useEffect(() => {
    const video = ref.current;
    if (!video) return;

    // iOS/autoplay safe
    video.muted = true;
    video.playsInline = true;
    video.preload = "auto";

    // reduced motion: no scrub, solo loop
    if (prefersReducedMotion()) {
      video.loop = true;
      video.autoplay = true;
      video.play().catch(() => {});
      return;
    }

    let st: ScrollTrigger | null = null;

    const onLoaded = () => {
      // importante: pausar para que scroll controle currentTime
      video.pause();
      video.currentTime = 0;

      // GSAP animando currentTime según el progreso del scroll
      const tl = gsap.timeline({ defaults: { ease: "none" } });
      tl.to(video, { currentTime: video.duration }, 0);

      st = ScrollTrigger.create({
        trigger,
        start,
        end,
        scrub: true,
        animation: tl,
        invalidateOnRefresh: true,
      });

      ScrollTrigger.refresh();
    };

    video.addEventListener("loadedmetadata", onLoaded);

    return () => {
      video.removeEventListener("loadedmetadata", onLoaded);
      st?.kill();
    };
  }, [src, trigger, start, end]);

  return (
    <video
      ref={ref}
      className={className}
      src={src}
      poster={poster}
      muted
      playsInline
      preload="auto"
    />
  );
}

import { useEffect, useState } from "react";

export function HUD() {
  const [p, setP] = useState(0);

  useEffect(() => {
    const onScroll = () => {
      const h = document.documentElement;
      const max = h.scrollHeight - h.clientHeight;
      setP(max > 0 ? (h.scrollTop / max) : 0);
    };
    onScroll();
    window.addEventListener("scroll", onScroll, { passive: true });
    return () => window.removeEventListener("scroll", onScroll);
  }, []);

  return (
    <div style={{ position: "fixed", top: 0, left: 0, right: 0, height: 2, zIndex: 60 }}>
      <div style={{
        height: "100%",
        width: `${p * 100}%`,
        background: "var(--accent)",
        opacity: 0.65
      }} />
    </div>
  );
}

14) Nota clave sobre “97 frames” y suavidad
------------------------------------------------------------

- No necesitas saber FPS: el scrub usa video.duration.
- Si el video “salta”:
  - aumenta preload y espera loadedmetadata (ya lo hacemos)
  - prueba exportar H.264 + yuv420p
- Si quieres “más suave” con Lenis:
  - usa end "+=1400" o "+=1800" para más recorrido de scroll => más control fino del currentTime.
